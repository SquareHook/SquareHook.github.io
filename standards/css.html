CSS Coding Standards

Indentation 

    
    /* Indent by 2 spaces at a time. Don't use tabs or mix tabs and spaces. */
   .example {
     color: blue;
    }

Formatting

    /* Avoid unnecessary qualifying ID and class names with type selectors */
    /*Not recommended */
    ul#example { }
    div.error { }
    
    /* Recommended */
    #example { }
    .error { }
    
    /* Use shorthand properties where possible */
    /* Not recommended */
    border-top-style: none;
    font-family: palatino, georgia, serif;
    font-size: 100%;
    line-height: 1.6;
    padding-bottom: 2em;
    padding-left: 1em;
    padding-right: 1em;
    padding-top: 0;
    
    /* Recommended */
    border-top: 0;
    font: 100%/1.6 palatino, georgia, serif;
    padding: 0 1em 2em;
    
    /* Declare CSS attributes in alphabetical order whenever possible */
    background: fuchsia;
    border: 1px solid;
    -moz-border-radius: 4px;
    -webkit-border-radius: 4px;
    border-radius: 4px;
    color: black;
    text-align: center;
    text-indent: 2em;
